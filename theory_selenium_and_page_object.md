class LoginPage:
    # –õ–æ–∫–∞—Ç–æ—Ä—ã —ç–ª–µ–º–µ–Ω—Ç–æ–≤

    ERROR_MESSAGE = (By.CLASS_NAME, "error")

        self.driver = driver
        self.driver.get("https://example.com/login")

        self.driver.find_element(*self.PASSWORD_INPUT).send_keys(password)
